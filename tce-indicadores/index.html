Configurar integração
Conecte o painel TCE ao Google Sheets do SGE
URL do Web App (Apps Script)
Cole aqui a URL gerada ao publicar o Apps Script do SGE como Web App. Formato: https://script.google.com/macros/s/SEU_ID/exec
Adicionar escolas da rede
Cadastre as escolas municipais de Juruá. O nome deve ser idêntico ao usado no SGE.
Como publicar o Apps Script
Siga os passos abaixo para configurar o Apps Script do SGE para expor os dados necessários ao painel TCE.
// 1. Abra o Google Sheets do SGE // 2. Extensões → Apps Script // 3. Adicione a função abaixo ao seu script existente // 4. Implante → Nova implantação → Web app // Executar como: Eu mesmo // Quem tem acesso: Qualquer pessoa // 5. Cole a URL gerada no campo acima function doGet(e) { var aba = e.parameter.aba || 'Frequencia'; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(aba); if (!sheet) return resposta({error: 'Aba não encontrada: ' + aba}); var dados = sheet.getDataRange().getValues(); var cabecalho = dados[0]; var linhas = dados.slice(1).map(function(row) { var obj = {}; cabecalho.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return resposta({aba: aba, dados: linhas}); } function resposta(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
Estrutura esperada nas abas do SGE
O Apps Script lerá as seguintes abas do seu Google Sheets. Verifique se os nomes das colunas correspondem:
Aba "Frequencia": data, escola, turma, aluno_id, aluno_nome, presente (S/N) Aba "Notas": escola, turma, aluno_id, aluno_nome, bimestre, disciplina, nota Aba "Alunos": escola, aluno_id, aluno_nome, turma, segmento, ativo (S/N), data_saida Aba "Diagnostico": escola, aluno_id, bimestre, nivel_leitura (1-4), nivel_escrita (1-4)
